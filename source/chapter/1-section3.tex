\section{Die Fouriertransformation auf $L^{1}$, $\Sz$ und $L^{2}$}



\begin{definition}
	Für $f \in L^{1}(\R^{d})$ setze $\hat{f}(\xi) = \mathcal{f}(\xi) = \int_{\R^{d}} e^{-2\pi i x \cdot \xi} f(x) dx$
\end{definition}


\begin{bemerkung*}
 	Für $d = 1$, $f \in L^{2}[0, \pi] \subseteq L^{2}(\R)$	
\end{bemerkung*}
